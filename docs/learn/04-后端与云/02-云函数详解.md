# 第四章 · 云函数详解

行迹共有 8 个云函数，本章逐一解读它们的作用和实现思路。

## 云函数通用结构

所有云函数都长这个样子：

```javascript
const cloud = require('wx-server-sdk')
cloud.init({ env: cloud.DYNAMIC_CURRENT_ENV })

const db = cloud.database()

exports.main = async (event, context) => {
  // event.xxx = 前端传来的参数
  // context  = 运行环境

  // 获取当前用户的 openid（无法伪造）
  const { OPENID } = cloud.getWXContext()

  try {
    // ... 业务逻辑
    return { ok: true, data: result }
  } catch (err) {
    console.error(err)
    return { error: err.message }
  }
}
```

## 1. ensure-user（登录/注册）

**作用**：小程序启动时调用，确保用户在数据库里有记录。

**流程**：
```
前端启动
  ↓ 调用 ensure-user
检查 users 集合里有没有这个 openid
  ↓ 有：更新 lastSeenAt
  ↓ 没有：创建新记录
返回 { openid, nickname, avatarUrl }
```

**代码逻辑**：

```javascript
exports.main = async (event, context) => {
  const { OPENID } = cloud.getWXContext()
  const { nickname, avatarUrl } = event

  const res = await db.collection('users').doc(OPENID).get().catch(() => null)

  if (res && res.data) {
    // 用户已存在，更新最后登录时间
    await db.collection('users').doc(OPENID).update({
      data: { lastSeenAt: db.serverDate() }
    })
    return { openid: OPENID, nickname: res.data.nickname, avatarUrl: res.data.avatarUrl }
  } else {
    // 新用户，创建记录
    await db.collection('users').doc(OPENID).set({
      data: { _openid: OPENID, nickname, avatarUrl, lastSeenAt: db.serverDate() }
    })
    return { openid: OPENID, nickname, avatarUrl }
  }
}
```

## 2. list-my-trips（获取攻略列表）

**作用**：获取用户所有相关攻略（自己创建的 + 协作的）。

**为什么要云函数而不是直接查？**  
因为「我参与协作的攻略」不属于我创建，直接查只能查到我创建的。云函数有管理员权限，可以查任何数据。

**代码逻辑**：

```javascript
exports.main = async (event, context) => {
  const { OPENID } = cloud.getWXContext()

  // 查两类：我创建的 OR 我是协作者的
  const result = await db.collection('trips')
    .where(db.command.or([
      { _openid: OPENID },
      { collaboratorOpenids: db.command.elemMatch(db.command.eq(OPENID)) }
    ]))
    .orderBy('updatedAt', 'desc')  // 按更新时间倒序
    .limit(100)
    .get()

  return { trips: result.data }
}
```

## 3. update-trip（保存攻略修改）

**作用**：保存对攻略的修改（由前端 500ms 防抖后调用）。

**关键设计**：白名单字段限制 + 权限检查

```javascript
const ALLOWED_FIELDS = ['name', 'pax', 'startDate', 'endDate', 'destinations', 'days', 'packing']

exports.main = async (event, context) => {
  const { OPENID } = cloud.getWXContext()
  const { tripId, patch } = event

  // 1. 白名单：只允许更新指定字段（防止恶意修改 _openid 等）
  const safeData = {}
  for (const key of ALLOWED_FIELDS) {
    if (key in patch) safeData[key] = patch[key]
  }

  // 2. 权限检查：是 owner 或 collaborator 才能改
  const trip = await db.collection('trips').doc(tripId).get()
  const isOwner = trip.data._openid === OPENID
  const isCollab = (trip.data.collaboratorOpenids || []).includes(OPENID)

  if (!isOwner && !isCollab) {
    return { error: '没有权限' }
  }

  // 3. 保存修改
  await db.collection('trips').doc(tripId).update({
    data: {
      ...safeData,
      updatedAt: db.serverDate(),  // 自动记录更新时间
      updatedBy: OPENID,           // 记录是谁改的
    }
  })

  return { ok: true }
}
```

## 4. create-share-token（生成分享链接）

**作用**：生成一个 7 天有效的分享 Token，用于分享攻略。

```javascript
const crypto = require('crypto')

exports.main = async (event, context) => {
  const { OPENID } = cloud.getWXContext()
  const { tripId, kind } = event  // kind: 'readonly' | 'collab'

  // 只有 owner 才能分享
  const trip = await db.collection('trips').doc(tripId).get()
  if (trip.data._openid !== OPENID) {
    return { error: '只有创建者才能分享' }
  }

  // 生成随机 Token（16字节十六进制 = 32个字符）
  const token = crypto.randomBytes(16).toString('hex')

  // 7 天后过期
  const expiresAt = new Date(Date.now() + 7 * 24 * 60 * 60 * 1000)

  // 存到数据库
  await db.collection('share_tokens').add({
    data: {
      token,
      tripId,
      kind,
      createdBy: OPENID,
      expiresAt: expiresAt.toISOString(),
      usedBy: [],
    }
  })

  return { token }
}
```

生成的 Token 会拼接到分享链接里：
```
/pages/share/index?token=abc123def456&kind=readonly&tripId=trip001
```

## 5. clone-trip（复制攻略）

**作用**：通过 readonly 分享链接，把别人的攻略复制一份到自己账号下。

```javascript
exports.main = async (event, context) => {
  const { OPENID } = cloud.getWXContext()
  const { sourceTripId, token } = event

  // 1. 验证 Token 有效性
  const tokenRes = await db.collection('share_tokens')
    .where({ token, tripId: sourceTripId, kind: 'readonly' })
    .get()

  const tokenDoc = tokenRes.data[0]
  if (!tokenDoc) return { error: 'Token 无效' }
  if (new Date(tokenDoc.expiresAt) < new Date()) return { error: 'Token 已过期' }

  // 2. 获取原攻略
  const sourceTrip = (await db.collection('trips').doc(sourceTripId).get()).data

  // 3. 创建副本（去掉原来的身份信息）
  const { _id, _openid, ownerOpenid, collaborators, collaboratorOpenids, createdAt, updatedAt, updatedBy, ...tripData } = sourceTrip

  const newTrip = {
    ...tripData,
    name: tripData.name + '（副本）',
    _openid: OPENID,
    ownerOpenid: OPENID,
    collaborators: [],
    collaboratorOpenids: [],
    createdAt: new Date().toISOString(),
    updatedAt: new Date().toISOString(),
  }

  const result = await db.collection('trips').add({ data: newTrip })

  // 4. 记录谁使用了这个 Token
  await db.collection('share_tokens').doc(tokenDoc._id).update({
    data: { usedBy: db.command.push(OPENID) }
  })

  return { newTripId: result._id }
}
```

## 6. join-collab（加入协作）

**作用**：通过 collab 分享链接，加入别人的攻略成为协作者。

```javascript
exports.main = async (event, context) => {
  const { OPENID } = cloud.getWXContext()
  const { tripId, token } = event

  // 验证 Token
  const tokenDoc = await validateToken(token, tripId, 'collab')
  if (!tokenDoc) return { error: 'Token 无效或过期' }

  // 获取攻略
  const trip = (await db.collection('trips').doc(tripId).get()).data

  // 已经是 owner？
  if (trip._openid === OPENID) return { ok: true, alreadyOwner: true }

  // 已经是协作者？
  if ((trip.collaboratorOpenids || []).includes(OPENID)) {
    return { ok: true, alreadyJoined: true }
  }

  // 获取用户信息
  const user = (await db.collection('users').doc(OPENID).get()).data

  // 加入协作者列表
  await db.collection('trips').doc(tripId).update({
    data: {
      collaborators: db.command.push({
        openid: OPENID,
        nickname: user.nickname,
        avatarUrl: user.avatarUrl,
        role: 'editor',
        joinedAt: new Date().toISOString(),
      }),
      collaboratorOpenids: db.command.push(OPENID),  // 同步更新快速查询字段
    }
  })

  return { ok: true }
}
```

## 7. amap-poi-search（地点搜索）

**作用**：搜索地点（调高德地图 API）。

**为什么不直接从前端调高德 API？**  
因为高德 API Key 放在前端代码里不安全（会被人盗用）。通过云函数代理，Key 只存在服务器上。

```javascript
const axios = require('axios')

exports.main = async (event) => {
  const { keyword, city } = event
  const AMAP_KEY = process.env.AMAP_KEY  // 从环境变量读 Key

  const url = 'https://restapi.amap.com/v5/place/text'
  const response = await axios.get(url, {
    params: {
      key: AMAP_KEY,
      keywords: keyword,
      city: city || '',
      page_size: 20,
    }
  })

  const pois = response.data.pois || []
  const results = pois.map(poi => ({
    name: poi.name,
    address: poi.address,
    city: poi.cityname,
    adcode: poi.adcode,
    lat: parseFloat(poi.location?.split(',')[1]),
    lng: parseFloat(poi.location?.split(',')[0]),
  }))

  return { results }
}
```

## 8. amap-weather（获取天气）

**作用**：根据城市代码（adcode）获取天气预报。

```javascript
exports.main = async (event) => {
  const { adcode } = event
  const AMAP_KEY = process.env.AMAP_KEY

  const response = await axios.get('https://restapi.amap.com/v3/weather/weatherInfo', {
    params: {
      key: AMAP_KEY,
      city: adcode,
      extensions: 'all',  // 获取预报（而不是当前天气）
    }
  })

  const forecasts = response.data.forecasts?.[0]?.casts
  if (!forecasts || forecasts.length === 0) return { weather: null }

  // 取明天（第一个预报）的天气
  const tomorrow = forecasts[0]
  return {
    weather: {
      city: response.data.forecasts[0].city,
      high: parseInt(tomorrow.daytemp),
      low: parseInt(tomorrow.nighttemp),
      description: tomorrow.dayweather,
      icon: mapWeatherIcon(tomorrow.dayweather),
    }
  }
}
```

## 云函数的错误处理

所有云函数都应该做错误处理，避免前端收到奇怪的错误：

```javascript
exports.main = async (event, context) => {
  try {
    // 业务逻辑
    return { ok: true, data: result }
  } catch (err) {
    console.error('[云函数错误]', err)
    // 返回友好的错误信息，不要把内部错误暴露给前端
    return { error: '操作失败，请重试' }
  }
}
```

## 小结

| 云函数 | 调用时机 | 核心操作 |
|--------|---------|---------|
| `ensure-user` | 小程序启动 | 创建/更新用户记录 |
| `list-my-trips` | 首页加载 | 查询我的所有攻略 |
| `update-trip` | 内容修改后 500ms | 保存攻略到数据库 |
| `create-share-token` | 点「分享」时 | 生成 Token |
| `clone-trip` | 接受只读分享 | 复制攻略 |
| `join-collab` | 接受协作分享 | 加入协作 |
| `amap-poi-search` | 搜索地点 | 查高德 API |
| `amap-weather` | 查看攻略 | 查高德天气 API |
